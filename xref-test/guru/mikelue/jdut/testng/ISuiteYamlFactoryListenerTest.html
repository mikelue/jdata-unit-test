15  @Test(suiteName="ISuiteYamlFactoryListenerTestSuite")
16  public class ISuiteYamlFactoryListenerTest extends AbstractDataSourceTestBase {
17  	private final static ISuiteYamlFactoryListener testedListener = new ISuiteYamlFactoryListener();
18  
19  	public ISuiteYamlFactoryListenerTest() {}
20  
21  	@BeforeMethod
22  	void setupTable() throws SQLException
23  	{
24  		JdbcTemplateFactory.buildRunnable(
25  			() -> getDataSource().getConnection(),
26  			conn -> DbStatement.buildRunnableForStatement(
27  				conn, stmt -> {
28  					stmt.execute("DROP TABLE IF EXISTS suite_t1");
29  					stmt.execute("CREATE TABLE suite_t1(t1_id INTEGER)");
30  				}
31  			).runJdbc()
32  		).runJdbc();
33  	}
34  
35  	/**
36  	 * Tests the suite processing.
37  	 */
38  	@Test
39  	public void suiteTest(ITestContext testContext) throws SQLException
40  	{
41  		YamlFactoryListenerBase.setDataSource(testContext.getSuite(), getDataSource());
42  
43  		testedListener.onStart(testContext.getSuite());
44  		assertData(1);
45  
46  		testedListener.onFinish(testContext.getSuite());
47  		assertData(2);
48  
49  		YamlFactoryListenerBase.removeDataSource(testContext.getSuite());
50  	}
51  
52  	private void assertData(int expectedCount) throws SQLException
53  	{
54  		String checkData = "SELECT COUNT(*) FROM suite_t1 WHERE t1_id = 27";
55  
56  		JdbcTemplateFactory.buildRunnable(
57  			() -> getDataSource().getConnection(),
58  			conn -> DbResultSet.buildRunnable(
59  				conn, checkData,
60  				rs -> new ResultSetAssert(rs)
61  					.assertNextTrue()
62  					.assertInt(1, expectedCount)
63  			).runJdbc()
64  		).runJdbc();
65  	}
