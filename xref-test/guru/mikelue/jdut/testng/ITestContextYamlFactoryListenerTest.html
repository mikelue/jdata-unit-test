17  @Test(testName="ITestContextYamlFactoryListenerTest")
18  @Listeners(ITestContextYamlFactoryListenerTest.DedicatedListener.class)
19  public class ITestContextYamlFactoryListenerTest extends AbstractDataSourceTestBase {
20  	public static class DedicatedListener extends ITestContextYamlFactoryListener {
21  		private AnnotationConfigApplicationContext ctx;
22  
23  		@Override
24  		public void onStart(ITestContext context)
25  		{
26  			if (ctx != null) {
27  				return;
28  			}
29  
30  			ctx = new AnnotationConfigApplicationContext();
31  			ctx.register(DataSourceContext.class);
32  			ctx.refresh();
33  
34  			YamlFactoryListenerBase.setDataSource(context, ctx.getBean(DataSource.class));
35  
36  			super.onStart(context);
37  		}
38  		@Override
39  		public void onFinish(ITestContext context)
40  		{
41  			super.onFinish(context);
42  
43  			YamlFactoryListenerBase.removeDataSource(context);
44  
45  			ctx.close();
46  			ctx = null;
47  		}
48  
49  		@Override
50  		protected boolean needConductData(ITestContext context)
51  		{
52  			return "ITestContextYamlFactoryListenerTest".equals(context.getName());
53  		}
54  	}
55  
56  	public ITestContextYamlFactoryListenerTest() {}

58  	/**
59  	 * Tests the listener for {@link ITestContext} of building/cleaning data.
60  	 */

64  		JdbcTemplateFactory.buildRunnable(
65  			() -> getDataSource().getConnection(),
66  			conn -> DbResultSet.buildRunnable(
67  				conn, "SELECT COUNT(*) FROM tcontext_t1",
68  				rs -> new ResultSetAssert(rs)
69  					.assertNextTrue()
70  					.assertInt(1, 0)
71  			).runJdbc()
72  		).runJdbc();
73  	}
