22  public class DefaultOperatorFactoryTest {
23  	@Mocked
24  	private DatabaseMetaData mockMetaData;
25  	@Mocked
26  	private DataSource mockDataSource;
27  
28  	public DefaultOperatorFactoryTest() {}
29  
30  	/**
31  	 * Tests the building of object and get specific matched operators.<br>
32  	 *
33  	 * The testing of default ones is performed by {@link DefaultOperatorsTest}.
34  	 */
35  	@ParameterizedTest
36  	@MethodSource
37  	public void build(
38  		String sampleProductName, String sampleOperatorName,
39  		Consumer<CallingTracer> assertion
40  	) throws SQLException {
41  		new Expectations() {{
42  			mockDataSource.getConnection().getMetaData();
43  			result = mockMetaData;
44  
45  			mockMetaData.getDatabaseProductName();
46  			result = sampleProductName;
47  		}};
48  
49  		CallingTracer tracer = new CallingTracer();
50  
51  		/**
52  		 * Prepare factory with customized operators
53  		 */
54  		OperatorFactory testedFactory = DefaultOperatorFactory.build(
55  			mockDataSource,
56  			builder -> builder
57  				.add(
58  					metaData -> metaData.getDatabaseProductName().equals("match-1"),
59  					buildSampleOperators(tracer::grainInsert1, tracer::grainUpdate)
60  				)
61  				.add( // Not used predicate
62  					metaData -> metaData.getDatabaseProductName().equals("match-1"),
63  					buildSampleOperators(tracer::grainInsert2, tracer::grainUpdate)
64  				)
65  		);
66  		// :~)
67  
68  		testedFactory.get(sampleOperatorName).operate(null, null);
69  
70  		assertion.accept(tracer);
71  	}
72  	static Arguments[] build()
73  	{
74  		return new Arguments[] {
75  			/**
76  			 * Asserts the matched operations
77  			 */
78  			arguments( "match-1", DefaultOperators.INSERT,
79  				(Consumer<CallingTracer>)tracer -> {
80  					assertTrue(tracer.insert1);
81  					assertFalse(tracer.insert2);
82  					assertFalse(tracer.update);
83  				}
84  			),
85  			arguments( "match-1", DefaultOperators.UPDATE,
86  				(Consumer<CallingTracer>)tracer -> {
87  					assertFalse(tracer.insert1);
88  					assertFalse(tracer.insert2);
89  					assertTrue(tracer.update);
90  				}
91  			),
92  			// :~)
93  		};
94  	}
95  	private Map<String, DataGrainOperator> buildSampleOperators(
96  		DataGrainOperator insertOperator, DataGrainOperator updateOperator
97  	) {
98  		Map<String, DataGrainOperator> operators = new HashMap<>();
99  		operators.put(DefaultOperators.INSERT, insertOperator);
100 		operators.put(DefaultOperators.UPDATE, updateOperator);
101 		return operators;
102 	}

106 /**
107  * Used for tracing the calling of lambda of DataGrainOperator.
108  */
109 class CallingTracer {
110 	boolean insert1 = false;
111 	boolean insert2 = false;
112 	boolean update = false;
113 
114 	DataGrain grainInsert1(Connection conn, DataGrain dataGrain)
115 	{
116 		insert1 = true;
117 		return dataGrain;
118 	}
119 	DataGrain grainInsert2(Connection conn, DataGrain dataGrain)
120 	{
121 		insert2 = true;
122 		return dataGrain;
123 	}
124 	DataGrain grainUpdate(Connection conn, DataGrain dataGrain)
125 	{
126 		update = true;
127 		return dataGrain;
128 	}
