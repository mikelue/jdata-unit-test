16  import guru.mikelue.jdut.jdbc.JdbcTemplateFactory;

18  import guru.mikelue.jdut.vendor.DatabaseVendor;
19  import snaq.db.DBPoolDataSource;
20  
21  @Configuration
22  @PropertySource("classpath:/test-context.properties")
23  public class DataSourceContext {
24  	private Logger logger = LoggerFactory.getLogger(DataSourceContext.class);
25  
26  	public DataSourceContext() {}
27  
28  	@Autowired
29  	private Environment env;
30  
31  	@Bean(destroyMethod="release")
32  	DataSource buildDataSource()
33  	{
34  		DBPoolDataSource dbPool = new DBPoolDataSource();
35  
36  		dbPool.setMaxPool(1);
37  		dbPool.setMaxSize(8);
38  		dbPool.setDriverClassName(env.getProperty("db.driverClassName"));
39  		dbPool.setUrl(env.getProperty("db.url"));
40  		dbPool.setUser(env.getProperty("db.username"));
41  		dbPool.setPassword(env.getProperty("db.password"));
42  
43  		logger.info(
44  			"Db connection: \"{}\". Driver: \"{}\"",
45  			env.getProperty("db.url"),
46  			env.getProperty("db.driverClassName")
47  		);
48  
49  		return dbPool;
50  	}
51  
52  	@Bean
53  	DbSchemaSetup buildDbSchemaSetup()
54  	{
55  		logger.warn("Db Schema Setup building ...");
56  		return new DbSchemaSetup();
57  	}
58  
59  	@Bean
60  	DatabaseVendor buildVender(
61  		DataSource dataSource
62  	) {
63  		return DatabaseVendor.getVendor(dataSource);
64  	}
65  
66  	public static PropertySourcesPlaceholderConfigurer buildPropertyPlaceholder()
67  	{
68  		return new PropertySourcesPlaceholderConfigurer();
69  	}
70  }
71  
72  class DbSchemaSetup {
73  	@Autowired
74  	private DataSource dataSource;
75  
76  	private Logger logger = LoggerFactory.getLogger(DbSchemaSetup.class);
77  
78  	private final static String TABLE_ARTIST =
79  		"CREATE TABLE IF NOT EXISTS ex_artist(" +
80  		"	at_id IDENTITY PRIMARY KEY," +
81  		"	at_name VARCHAR(512) NOT NULL," +
82  		"	at_gender TINYINT NOT NULL DEFAULT 3," +
83  		"	at_birthday DATE" +
84  		")";
85  	private final static String TABLE_ALBUM =
86  		"CREATE TABLE IF NOT EXISTS ex_album(" +
87  		"	ab_id IDENTITY PRIMARY KEY," +
88  		"	ab_name VARCHAR(512) NOT NULL," +
89  		"	ab_release_date DATE NOT NULL," +
90  		"	ab_duration_seconds SMALLINT NOT NULL," +
91  		"	ab_type TINYINT NOT NULL DEFAULT 1," +
92  		"	ab_at_id INTEGER NOT NULL" +
93  		")";
94  
95  	@EventListener
96  	public void setupSchema(ContextRefreshedEvent e)
97  	{
98  		logger.warn("[Schema] setup");
99  
100 		JdbcTemplateFactory.buildRunnable(
101 			() -> dataSource.getConnection(),
102 			conn -> DbStatement.buildRunnableForStatement(
103 				conn,
104 				stat -> {
105 					stat.executeUpdate(TABLE_ALBUM);
106 					stat.executeUpdate(TABLE_ARTIST);
107 				}
108 			).runJdbc()
109 		).asRunnable().run();
110 	}
111 }
