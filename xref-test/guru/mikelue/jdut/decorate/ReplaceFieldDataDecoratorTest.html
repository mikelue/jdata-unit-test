9   public class ReplaceFieldDataDecoratorTest {
10  	public ReplaceFieldDataDecoratorTest() {}
11  
12  	/**
13  	 * Tests the replacement decorator.
14  	 */
15  	@Test
16  	public void build()
17  	{
18  		DataGrainDecorator testedDecorator = ReplaceFieldDataDecorator.buildDataGrainDecorator(
19  			builder -> builder
20  				.replaceWith(dataField -> dataField.getColumnName().equals("gc_1"), "MT-01")
21  				.replaceWith("[V1]", 20)
22  				.replaceWith("[NULL]", null) // Replaces to null value
23  				.replaceWith("[V3]", 100) // Unused
24  		);
25  
26  		DataGrain sampleDataGrain = DataGrain.build(
27  			tableSchema -> tableSchema.name("tab_008"),
28  			rowsBuilder -> rowsBuilder
29  				.implicitColumns("gc_1", "gc_2", "gc_3", "gc_4")
30  				.addValues("MT-UN-01", "[V1]", 98, "[NULL]")
31  		).decorate(testedDecorator);
32  
33  		assertEquals("MT-01", sampleDataGrain.getRow(0).getData("gc_1"));
34  		assertEquals(Integer.valueOf(20), sampleDataGrain.getRow(0).getData("gc_2"));
35  		assertEquals(Integer.valueOf(98), sampleDataGrain.getRow(0).getData("gc_3"));
36  		assertEquals((Integer)null, sampleDataGrain.getRow(0).getData("gc_4"));
37  	}
