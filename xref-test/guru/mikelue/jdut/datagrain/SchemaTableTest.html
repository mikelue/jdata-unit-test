24  	/**
25  	 * Tests the process for quoted full name
26  	 */
27  	@ParameterizedTest
28  	@MethodSource
29  	void buildForQuotedFullName(
30  		boolean metaDataWorker,
31  		String sampleSchema, String sampleTable,
32  		String expectedResult,
33  		@Mocked MetaDataWorker mockMetaDataWorker
34  	) {
35  		if (metaDataWorker) {
36  			new Expectations() {{
37  				mockMetaDataWorker.processIdentifier(anyString);
38  				result = new Delegate<Object>() {
39  					@SuppressWarnings("unused")
40  					String processIdentifier(String source)
41  					{
42  						return source;
43  					}
44  				};
45  
46  				mockMetaDataWorker.quoteIdentifier(anyString);
47  				result = new Delegate<Object>() {
48  					@SuppressWarnings("unused")
49  					String quoteIdentifier(String source)
50  					{
51  						return String.format("\"%s\"", source);
52  					}
53  				};
54  
55  				mockMetaDataWorker.supportsSchemasInDataManipulation();
56  				result = true;
57  			}};
58  		}
59  
60  		SchemaTable testedTableSchema = SchemaTable.build(
61  			builder -> builder
62  				.metaDataWorker(metaDataWorker ? mockMetaDataWorker : null)
63  				.schema(sampleSchema)
64  				.name(sampleTable)
65  		);
66  
67  		assertThat(testedTableSchema.getQuotedFullName())
68  			.isEqualTo(expectedResult);
69  	}
70  	static Arguments[] buildForQuotedFullName()
71  	{
72  		return new Arguments[] {
73  			arguments(true, "", "hammer", "\"hammer\""),
74  			arguments(true, "black", "mallet", "\"black\".\"mallet\""),
75  			arguments(false, "", "hammer", "hammer"),
76  			arguments(false, "black", "mallet", "mallet"),
77  		};
78  	}
