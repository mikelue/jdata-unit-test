3   import java.lang.annotation.*;
4   import static java.lang.annotation.ElementType.*;
5   
6   import guru.mikelue.jdut.vendor.DatabaseVendor;
7   
8   /**
9    * Defines the annotation used to check if the current vendor is matched the environment(implemented by container, framework., etc.).<br>
10   *
11   * The default value is {@link DatabaseVendor#Unknown}, which means nothing to checked(everything is passed always).
12   */
13  @Target({ANNOTATION_TYPE, CONSTRUCTOR, FIELD, LOCAL_VARIABLE, METHOD, PACKAGE, PARAMETER, TYPE, TYPE_PARAMETER, TYPE_USE})
14  @Retention(RetentionPolicy.RUNTIME)
15  @Documented @Inherited
16  public @interface IfDatabaseVendor {
17  	/**
18  	 * Any one of the vendors matches environment.<br>
19  	 *
20  	 * Default value is {@link DatabaseVendor#Unknown}
21  	 *
22  	 * @return The array of vendors that any one of them is matched
23  	 */
24  	public DatabaseVendor[] match() default DatabaseVendor.Unknown;
25  	/**
26  	 * Any one of the vendors must not match environment.<br>
27  	 *
28  	 * Default value is {@link DatabaseVendor#Unknown}
29  	 *
30  	 * @return The array of vendors that any one of them is not matched
31  	 */
32  	public DatabaseVendor[] notMatch() default DatabaseVendor.Unknown;
33  }
