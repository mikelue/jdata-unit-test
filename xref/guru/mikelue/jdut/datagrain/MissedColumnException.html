3   /**
4    * Represents the exception which defined data field cannot be found

6    */
7   public class MissedColumnException extends DataRowException {

11  
12  	public MissedColumnException(SchemaTable tableSchema, SchemaColumn newMissedColumn)
13  	{
14  		super(String.format("Miss column: \"%s.%s\"", tableSchema.getName(), newMissedColumn.getName()));
15  		missedColumn = newMissedColumn;
16  	}
17  
18  	public SchemaColumn getMissedColumn()
19  	{
20  		return missedColumn;
21  	}
