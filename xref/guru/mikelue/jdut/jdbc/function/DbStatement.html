19  /**
20   * Utility used to build {@link JdbcRunnable} or {@link JdbcSupplier}
21   * for processing a {@link Statement} or {@link PreparedStatement} with other dependencies(e.g., {@link DataRow}).
22   */
23  public final class DbStatement {
24  	private DbStatement() {}
25  
26  	/**
27  	 * Builds runnable to set-up {@link PreparedStatement} and fed it to void function.
28  	 *
29  	 * @param conn The connection of database(won't be closed in the lambda)
30  	 * @param sql The SQL for executing statement
31  	 * @param nameOfColumns The sequence of columns used to call setXXX() of statement
32  	 * @param dataRow The data to be set into statement
33  	 * @param executor The executor for built statement
34  	 *
35  	 * @return The instance of lambda
36  	 */
37  	public static JdbcRunnable buildRunnableForPreparedStatement(
38  		Connection conn,
39  		String sql, List<String> nameOfColumns,
40  		DataRow dataRow,
41  		JdbcVoidFunction<? super PreparedStatement> executor
42  	) {
43  		return JdbcTemplateFactory.buildRunnable(
44  			() -> conn.prepareStatement(sql),
45  			stat -> {
46  				nameOfColumns.forEach(
47  					new DataRowParameterSetter(stat, dataRow)
48  						.asConsumer()
49  				);
50  
51  				executor.applyJdbc(stat);
52  			}
53  		);
54  	}

58  	 *
59  	 * @param conn The connection of database(won't be closed in the lambda)
60  	 * @param sql The SQL for executing statement
61  	 * @param values The sequence of values used to call setXXX() of statement
62  	 * @param jdbcTypeMapping The type of JDBC mapping by type of data
63  	 * @param executor The executor for built statement
64  	 *
65  	 * @return The instance of lambda
66  	 *
67  	 * @see PreparedStatements#setParameter(PreparedStatement, Object, JDBCType, int)
68  	 */
69  	public static JdbcRunnable buildRunnableForPreparedStatement(
70  		Connection conn,
71  		String sql,
72  		List<Object> values, Function<Object, JDBCType> jdbcTypeMapping,
73  		JdbcVoidFunction<? super PreparedStatement> executor
74  	) {
75  		return JdbcTemplateFactory.buildRunnable(
76  			() -> conn.prepareStatement(sql),
77  			stat -> {
78  				values.forEach(
79  					new ValueParameterSetter(stat, jdbcTypeMapping)
80  						.asConsumer()
81  				);
82  
83  				executor.applyJdbc(stat);
84  			}
85  		);
86  	}

107 	) {

189 	private static class DataRowParameterSetter implements JdbcVoidFunction<String> {
190 		private PreparedStatement statement;
191 		private DataRow dataRow;
192 		private int paramIndex = 1;
193 
194 		private DataRowParameterSetter(
195 			PreparedStatement newStatement, DataRow newDataRow
196 		) {
197 			statement = newStatement;
198 			dataRow = newDataRow;
199 		}
200 
201 		@Override
202 		public void applyJdbc(String columnName) throws SQLException
203 		{
204 			PreparedStatements.setParameter(
205 				statement, dataRow, columnName,
206 				paramIndex++
207 			);
208 		}
209 	}
210 
211 	private static class ValueParameterSetter implements JdbcVoidFunction<Object> {
212 		private PreparedStatement statement;
213 		private Function<Object, JDBCType> jdbcTypeMapping;
214 		private int paramIndex = 1;
215 
216 		private ValueParameterSetter(
217 			PreparedStatement newStatement,
218 			Function<Object, JDBCType> newJdbcTypeMapping
219 		) {
220 			statement = newStatement;
221 			jdbcTypeMapping = newJdbcTypeMapping;
222 		}
223 
224 		@Override
225 		public void applyJdbc(Object data) throws SQLException
226 		{
227 			PreparedStatements.setParameter(
228 				statement, data, jdbcTypeMapping.apply(data), paramIndex++
229 			);
230 		}
231 	}
