3   import java.sql.SQLException;
4   import java.util.function.Function;

6   /**
7    * Converts the {@link SQLException} into another instance of {@link RuntimeException}.<p>
8    *
9    * @param <E> The type of exception to be generated
10   */
11  @FunctionalInterface
12  public interface SQLExceptionConvert<E extends RuntimeException> extends Function<SQLException, E> {
13  	/**
14  	 * The pre-defined instance for converting {@link SQLException} to {@link RuntimeException}.
15  	 *
16  	 * @param e The sql exception to be converted
17  	 *
18  	 * @return runtime exception
19  	 */
20  	public static RuntimeException runtimeException(SQLException e)
21  	{
22  		return new RuntimeException(e);
23  	}
24  }
