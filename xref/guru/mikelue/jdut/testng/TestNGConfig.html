3   import java.lang.annotation.*;

5   import org.testng.annotations.DataProvider;
6   
7   import static java.lang.annotation.ElementType.*;
8   
9   /**
10   * Defines the behaviours working by {@link IInvokedMethodYamlFactoryListener}.<br>
11   *
12   * With {@code @TestNGConfig(oneTimeOnly=true)}, the data conductor performs actions only on
13   * <strong>first time and last time</strong> of multiple tests on a testing method.
14   *
15   * <p>For example(by {@link DataProvider}):</p>
16   * <pre><code class="java">
17   * &#64;Test &#64;TestNGConfig(oneTimeOnly=true) &#64;JdutResource
18   * public void mixStuff(int speed)
19   * {
20   *     // Executes tested code
21   *     // Assertions ...
22   * }
23   * // Executes DuetConductor.build() before case of "0".
24   * // Executes DuetConductor.clean() after case of "1275".
25   * &#64;DataProvider
26   * Object[][] mixStuff()
27   * {
28   *     return new Object[][] {
29   *         { 0 }, { 30 }, { 2000 },
30   *         { -1 }, { 50000 }, { 1275 },
31   *     };
32   * }
33   * </code></pre>
34   */
35  @Target({ANNOTATION_TYPE, CONSTRUCTOR, FIELD, LOCAL_VARIABLE, METHOD, PACKAGE, PARAMETER, TYPE, TYPE_PARAMETER, TYPE_USE})
36  @Retention(RetentionPolicy.RUNTIME)
37  @Documented @Inherited
38  public @interface TestNGConfig {
39  	/**
40  	 * With using of {@link DataProvider}, set this field to true to build data only once
41  	 * before first invocation and after last invocation.
42  	 *
43  	 * @return true if conduction data one time only
44  	 */
45  	public boolean oneTimeOnly() default false;
46  }
