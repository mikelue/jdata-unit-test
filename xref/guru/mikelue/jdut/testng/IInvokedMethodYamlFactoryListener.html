19  /**
20   * This listener uses file name(<code style="color:blue">{@literal <class_name>-<method_name>.yaml}</code>) for resource({@link Class#getResource Class.getResource})
21   * of conducting data.
22   *
23   * <p>Only <b>testing method</b> which is annotated {@link JdutResource @JdutResource} would be prepared for data conduction.</p>
24   *
25   * <p>By default, the {@link DataSource} would be retrieved from {@link ITestContext} object(which type of {@link IAttributes}).</p>
26   *
27   * <p>By value of {@link TestNGConfig#oneTimeOnly}, this listener would executes only one time or multiple times
28   * for testing method with data provider.</p>
29   *
30   * <p>It is recommended that client implements {@link #needConductData} to trigger data conduction.</p>
31   *
32   * @see TestNGConfig
33   * @see AnnotationUtil#buildConductorByConvention(YamlConductorFactory, Method)
34   */
35  public class IInvokedMethodYamlFactoryListener extends YamlFactoryListenerBase implements IInvokedMethodListener {
36  	public IInvokedMethodYamlFactoryListener() {}
37  
38  	/**
39  	 * Executes {@link #duetConductorBuild duetConductorBuild} if the method is 1) testing method and 2) annotated with {@link JdutResource}.
40  	 *
41  	 * @param method The method provided by TestNG framework
42  	 * @param testResult The result object provided byh TestNG framework
43  	 */
44  	@Override
45  	public void beforeInvocation(IInvokedMethod method, ITestResult testResult)
46  	{
47  		if (!needConductData(method, testResult)) {
48  			getLogger().trace("[Build Data] Not matched context of method: [{}]",
49  				method
50  			);
51  			return;
52  		}
53  
54  		ITestNGMethod testNgMethod = method.getTestMethod();
55  
56  		if (!method.isTestMethod() ||
57  			!AnnotationUtil.hasJdutResourceAnnotation(testNgMethod.getConstructorOrMethod().getMethod())
58  		) {
59  			return;
60  		}
61  
62  		if (TestNGConfigUtil.oneTimeOnly(testNgMethod) && !TestNGConfigUtil.firstTime(testNgMethod)) {
63  			if (getLogger().isTraceEnabled()) {
64  				getLogger().trace("[Building Data] Not first time for method: [{}]",
65  					testNgMethod.getConstructorOrMethod().getMethod().getName()
66  				);
67  			}
68  			return;
69  		}
70  
71  		duetConductorBuild(testResult);
72  	}
73  	/**
74  	 * Executes {@link #duetConductorClean duetConductorClean} if the method is 1) testing method and 2) annotated with {@link JdutResource}.
75  	 *
76  	 * @param method The method provided by TestNG framework
77  	 * @param testResult The result object provided byh TestNG framework
78  	 */
79  	@Override
80  	public void afterInvocation(IInvokedMethod method, ITestResult testResult)
81  	{
82  		if (!needConductData(method, testResult)) {
83  			getLogger().trace("[Clean Data] Not matched context of method: [{}]",
84  				method
85  			);
86  			return;
87  		}
88  
89  		ITestNGMethod testNgMethod = method.getTestMethod();
90  
91  		if (!method.isTestMethod() ||
92  			!AnnotationUtil.hasJdutResourceAnnotation(testNgMethod.getConstructorOrMethod().getMethod())
93  		) {
94  			return;
95  		}
96  
97  		if (TestNGConfigUtil.oneTimeOnly(testNgMethod) && !TestNGConfigUtil.lastTime(testNgMethod)) {
98  			if (getLogger().isTraceEnabled()) {
99  				getLogger().trace("[Clean Data] Not last time for method: [{}]",
100 					testNgMethod.getConstructorOrMethod().getMethod().getName()
101 				);
102 			}
103 			return;
104 		}
105 
106 		duetConductorClean(testResult);
107 	}
108 
109 	/**
110 	 * Builds the conductor from data source in {@link ITestContext} of {@link ITestResult}.<br>
111 	 *
112 	 * This method would build conductor only for testing method with {@link JdutResource} annotation.<br>
113 	 *
114 	 * @param attributes The object of {@link ITestResult}
115 	 *
116 	 * @return The initialized conductor or nothing
117 	 */
118 	@Override
119 	protected Optional<DuetConductor> buildDuetConductor(IAttributes attributes)
120 	{
121 		ITestResult result = (ITestResult)attributes;
122 
123 		Method method = result.getMethod().getConstructorOrMethod().getMethod();
124 
125 		if (getLogger().isDebugEnabled()) {
126 			getLogger().debug(
127 				"Build DuetConductor by file name: \"{}\"",
128 				TestNgResourceNaming.naming(
129 					"{1}-{4}",
130 					result.getMethod(),
131 					".yaml"
132 				)
133 			);
134 		}
135 
136 		return AnnotationUtil.buildConductorByConvention(buildYamlConductorFactory(result.getTestContext()), method);
137 	}
138 
139 	/**
140 	 * Checks whether or not the context should make.<br>
141 	 *
142 	 * By default, this method always returns true.
143 	 *
144 	 * @param method The method provided by TestNG framework
145 	 * @param testResult The result object provided byh TestNG framework
146 	 *
147 	 * @return true if this context should apply data conduction
148 	 */
149 	protected boolean needConductData(IInvokedMethod method, ITestResult testResult)
150 	{
151 		return true;
152 	}
