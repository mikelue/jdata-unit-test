3   import java.util.Map;
4   import java.util.Optional;
5   import java.util.concurrent.ConcurrentHashMap;
6   import javax.sql.DataSource;
7   
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  import org.apache.commons.lang3.Validate;
11  import org.testng.IAttributes;
12  
13  import guru.mikelue.jdut.DuetConductor;
14  import guru.mikelue.jdut.yaml.YamlConductorFactory;
15  
16  /**
17   * Defines the basic, overridable method for constructing listeners.<br>
18   *
19   * <p>This listener uses {@link ConcurrentHashMap} for building mapping between {@link IAttributes} and {@link DuetConductor},
20   * so you should use the implementation of this listener in multi-thread environment with discretion.</p>
21   *
22   * <h3>Main methods</h3>
23   * <ul>
24   * 	<li>{@link #buildDuetConductor} - Client must be implementing this method</li>
25   * 	<li>{@link #buildDataSource} - Gets/build data source</li>
26   * 	<li>{@link #buildYamlConductorFactory} - Gets/builder singleton instance for {@link YamlConductorFactory}</li>
27   * </ul>
28   *
29   * @see ISuiteYamlFactoryListener
30   * @see ITestContextYamlFactoryListener
31   * @see IInvokedMethodYamlFactoryListener
32   */
33  public abstract class YamlFactoryListenerBase {
34  	private Map<IAttributes, Optional<DuetConductor>> conductors = new ConcurrentHashMap<>(4);
35  
36  	private Logger logger = LoggerFactory.getLogger(getClass());
37  	private static Logger staticLogger = LoggerFactory.getLogger(YamlFactoryListenerBase.class);
38  
39  	private final static String DATA_SOURCE = "_jdut_data_source_";
40  
41  	/**
42  	 * Sets the attribute for data source.
43  	 *
44  	 * @param attributes The attributes to be set
45  	 * @param dataSource The initialized data source
46  	 */
47  	public static void setDataSource(IAttributes attributes, DataSource dataSource)
48  	{
49  		staticLogger.debug("Set data source[{}] to IAttributes: [{}]", dataSource, attributes);
50  		attributes.setAttribute(DATA_SOURCE, dataSource);
51  	}
52  	/**
53  	 * Gets the data source from attribute
54  	 *
55  	 * @param attributes The attribute may contain data source
56  	 *
57  	 * @return The put data source
58  	 */
59  	public static DataSource getDataSource(IAttributes attributes)
60  	{
61  		DataSource dataSource = (DataSource)attributes.getAttribute(DATA_SOURCE);
62  		Validate.notNull(dataSource, "Cannot find data source");
63  
64  		return dataSource;
65  	}
66  	/**
67  	 * Removes the data source from attribute
68  	 *
69  	 * @param attributes The attribute may contain data source
70  	 *
71  	 * @return The data source put by {@link #setDataSource}
72  	 */
73  	public static DataSource removeDataSource(IAttributes attributes)
74  	{
75  		staticLogger.debug("Remove data source from IAttributes: [{}]", attributes);
76  		return (DataSource)attributes.removeAttribute(DATA_SOURCE);
77  	}
78  
79  	protected YamlFactoryListenerBase() {}
80  
81  	/**
82  	 * Gets data source by attribute.
83  	 *
84  	 * @param attributes The attributes object
85  	 *
86  	 * @return The initialized data source put from other place
87  	 *
88  	 * @see #setDataSource
89  	 * @see #getDataSource
90  	 */
91  	protected DataSource buildDataSource(IAttributes attributes)
92  	{
93  		return getDataSource(attributes);
94  	}
95  
96  	/**
97  	 * Uses {@link #buildDataSource} to build {@link YamlConductorFactory} or return a single instance of it.
98  	 *
99  	 * @param testResult This object contains both testing method and context
100 	 *
101 	 * @return The singleton factory of YAML
102 	 */
103 	private YamlConductorFactory yamlFactory = null;
104 	protected YamlConductorFactory buildYamlConductorFactory(IAttributes attributes)
105 	{
106 		if (yamlFactory == null) {
107 			logger.debug("Builds YamlConductorFactory by default implementation. IAttributes: [{}]", attributes);
108 			yamlFactory = YamlConductorFactory.build(
109 				getDataSource(attributes)
110 			);
111 		}
112 
113 		return yamlFactory;
114 	}
115 
116 	/**
117 	 * By {@link #buildDuetConductor}, executes {@link DuetConductor#build}.<br>
118 	 *
119 	 * The conductor would be a new one from {@link #buildDuetConductor}, which would be put into fed {@link IAttributes}.
120 	 *
121 	 * @param attributes The attribute object of TestNG
122 	 */
123 	protected final void duetConductorBuild(IAttributes attributes)
124 	{
125 		Optional<DuetConductor> conductor = buildDuetConductor(attributes);
126 		conductor.ifPresent(
127 			workingConductor -> workingConductor.build()
128 		);
129 
130 		logger.debug("Puts DuetConductor[{}] to IAttributes: [{}]", conductor, attributes);
131 
132 		conductors.put(attributes, conductor);
133 	}
134 	/**
135 	 * Executes {@link DuetConductor#clean} keep in {@link IAttributes}, which is saved by {@link #duetConductorBuild}.<br>
136 	 *
137 	 * The conductor would be cleaned.
138 	 *
139 	 * @param attributes The attribute object of TestNG
140 	 */
141 	protected final void duetConductorClean(IAttributes attributes)
142 	{
143 		if (!conductors.containsKey(attributes)) {
144 			return;
145 		}
146 
147 		Optional<DuetConductor> conductor = conductors.get(attributes);
148 		logger.debug("Remove DuetConductor[{}] from IAttributes: [{}]", conductor, attributes);
149 
150 		conductor.ifPresent(
151 			workingConductor -> workingConductor.clean()
152 		);
153 	}
154 
155 	/**
156 	 * Gets the logger with name of implementing class.
157 	 *
158 	 * @return The logger
159 	 */
160 	protected Logger getLogger()
161 	{
162 		return logger;
163 	}
164 
165 	/**
166 	 * Builds {@link DuetConductor}(as {@link Optional}), client should implements this method.
167 	 *
168 	 * @param attributes The attribute object used for retrieve testing environment
169 	 *
170 	 * @return The initialized conductor for testing
171 	 */
172 	protected abstract Optional<DuetConductor> buildDuetConductor(IAttributes attributes);
173 }
