21  /**
22   * Defines the basic, overridable method for constructing listeners.<br>
23   *
24   * <p>This listener uses {@link ConcurrentHashMap} for building mapping between {@link IAttributes} and {@link DuetConductor},
25   * so you should use the implementation of this listener in multi-thread environment with discretion.</p>
26   *
27   * The value of <strong>hash key</strong> on {@link IAttributes}:
28   * <ul>
29   *   <li>For {@link ITestContext} - Use suite name and test name</li>
30   *   <li>For {@link ISuite} - Use suite name</li>
31   *   <li>For {@link ITestResult} - Use canonical class name and method name</li>
32   * </ul>
33   *
34   * <h3>Main methods</h3>
35   * <ul>
36   * 	<li>{@link #buildDuetConductor} - Client must be implementing this method</li>
37   * 	<li>{@link #buildYamlConductorFactory} - Gets/builder singleton instance for {@link YamlConductorFactory}</li>
38   * </ul>
39   *
40   * @see ISuiteYamlFactoryListener
41   * @see ITestContextYamlFactoryListener
42   * @see IInvokedMethodYamlFactoryListener
43   */
44  public abstract class YamlFactoryListenerBase {
45  	private Map<Integer, Optional<DuetConductor>> conductors = new ConcurrentHashMap<>(4);
46  
47  	private Logger logger = LoggerFactory.getLogger(getClass());
48  	private static Logger staticLogger = LoggerFactory.getLogger(YamlFactoryListenerBase.class);
49  
50  	private final static String DATA_SOURCE = "_jdut_data_source_";
51  
52  	/**
53  	 * Sets the attribute for data source.
54  	 *
55  	 * @param attributes The attributes to be set
56  	 * @param dataSource The initialized data source
57  	 */
58  	public static void setDataSource(IAttributes attributes, DataSource dataSource)
59  	{
60  		if (staticLogger.isDebugEnabled()) {
61  			staticLogger.debug("Set data source[{}] to: [{}]", dataSource, formatIAttritubes(attributes));
62  		}
63  		attributes.setAttribute(DATA_SOURCE, dataSource);
64  	}
65  	/**
66  	 * Gets the data source from attribute
67  	 *
68  	 * @param attributes The attribute may contain data source
69  	 *
70  	 * @return The put data source
71  	 */
72  	public static DataSource getDataSource(IAttributes attributes)
73  	{
74  		if (staticLogger.isDebugEnabled()) {
75  			staticLogger.debug("Find data source from: {}", formatIAttritubes(attributes));
76  		}
77  		DataSource dataSource = (DataSource)attributes.getAttribute(DATA_SOURCE);
78  		Validate.notNull(dataSource, "Cannot find data source");
79  
80  		return dataSource;
81  	}
82  	/**
83  	 * Removes the data source from attribute
84  	 *
85  	 * @param attributes The attribute may contain data source
86  	 *
87  	 * @return The data source put by {@link #setDataSource}
88  	 */
89  	public static DataSource removeDataSource(IAttributes attributes)
90  	{
91  		if (staticLogger.isDebugEnabled()) {
92  			staticLogger.debug("Remove data source from: [{}]", formatIAttritubes(attributes));
93  		}
94  		return (DataSource)attributes.removeAttribute(DATA_SOURCE);
95  	}
96  
97  	protected YamlFactoryListenerBase() {}
98  
99  	/**
100 	 * Uses {@link #getDataSource} to build {@link YamlConductorFactory} or return a single instance of it.
101 	 *
102 	 * @param testResult This object contains both testing method and context
103 	 *
104 	 * @return The singleton factory of YAML
105 	 */
106 	private YamlConductorFactory yamlFactory = null;
107 	protected YamlConductorFactory buildYamlConductorFactory(IAttributes attributes)
108 	{
109 		if (yamlFactory == null) {
110 			if (logger.isDebugEnabled()) {
111 				logger.debug("Builds YamlConductorFactory by default implementation. Data source from: [{}]",
112 					formatIAttritubes(attributes));
113 			}
114 			yamlFactory = YamlConductorFactory.build(
115 				getDataSource(attributes)
116 			);
117 		}
118 
119 		return yamlFactory;
120 	}
121 
122 	/**
123 	 * By {@link #buildDuetConductor}, executes {@link DuetConductor#build}.<br>
124 	 *
125 	 * The conductor would be a new one from {@link #buildDuetConductor}, which would be put into fed {@link IAttributes}.
126 	 *
127 	 * @param attributes The attribute object of TestNG
128 	 */
129 	protected final void duetConductorBuild(IAttributes attributes)
130 	{
131 		Optional<DuetConductor> conductor = buildDuetConductor(attributes);
132 		conductor.ifPresent(
133 			workingConductor -> {
134 				if (logger.isDebugEnabled()) {
135 					logger.debug("Build data for: {}", formatIAttritubes(attributes));
136 				}
137 				workingConductor.build();
138 			}
139 		);
140 
141 		logger.debug("Puts DuetConductor to it");
142 
143 		conductors.put(buildContextKey(attributes), conductor);
144 	}
145 	/**
146 	 * Executes {@link DuetConductor#clean} keep in {@link IAttributes}, which is saved by {@link #duetConductorBuild}.<br>
147 	 *
148 	 * The conductor would be cleaned.
149 	 *
150 	 * @param attributes The attribute object of TestNG
151 	 */
152 	protected final void duetConductorClean(IAttributes attributes)
153 	{
154 		int contextKey = buildContextKey(attributes) ;
155 
156 		if (!conductors.containsKey(contextKey)) {
157 			logger.warn("No conductor in attribute: {}", formatIAttritubes(attributes));
158 			return;
159 		}
160 
161 		Optional<DuetConductor> conductor = conductors.get(contextKey);
162 
163 		conductor.ifPresent(
164 			workingConductor -> {
165 				if (logger.isDebugEnabled()) {
166 					logger.debug("Clean data for: {}", formatIAttritubes(attributes));
167 				}
168 				workingConductor.clean();
169 			}
170 		);
171 
172 		conductors.remove(contextKey);
173 		logger.debug("Remove DuetConductor from it");
174 	}
175 
176 	/**
177 	 * Gets the logger with name of implementing class.
178 	 *
179 	 * @return The logger
180 	 */
181 	protected Logger getLogger()
182 	{
183 		return logger;
184 	}
185 
186 	/**
187 	 * Builds {@link DuetConductor}(as {@link Optional}), client should implements this method.
188 	 *
189 	 * @param attributes The attribute object used for retrieve testing environment
190 	 *
191 	 * @return The initialized conductor for testing
192 	 */
193 	protected abstract Optional<DuetConductor> buildDuetConductor(IAttributes attributes);
194 
195 	private static int buildContextKey(IAttributes attr)
196 	{
197 		if (ITestContext.class.isInstance(attr)) {
198 			ITestContext context = (ITestContext)attr;
199 			return new HashCodeBuilder()
200 				.append(context.getSuite().getName())
201 				.append(context.getName())
202 				.toHashCode();
203 		} else if (ISuite.class.isInstance(attr)) {
204 			ISuite suite = (ISuite)attr;
205 			return new HashCodeBuilder()
206 				.append(suite.getName())
207 				.toHashCode();
208 		} else if (ITestResult.class.isInstance(attr)) {
209 			ITestNGMethod method = ((ITestResult)attr).getMethod();
210 
211 			return new HashCodeBuilder()
212 				.append(method.getRealClass().getCanonicalName())
213 				.append(method.getMethodName())
214 				.toHashCode();
215 		}
216 
217 		throw new IllegalArgumentException(String.format("Cannot figure out the HASH CODE of type: %s", attr.getClass()));
218 	}
219 	private static String formatIAttritubes(IAttributes attr)
220 	{
221 		if (ITestContext.class.isInstance(attr)) {
222 			ITestContext context = (ITestContext)attr;
223 			return String.format(
224 				"ITestContext >> Suite: [%s]. Test [%s].",
225 				context.getSuite().getName(),
226 				context.getName()
227 			);
228 		} else if (ISuite.class.isInstance(attr)) {
229 			ISuite suite = (ISuite)attr;
230 			return String.format(
231 				"ISuite >> Suite: [%s].", suite.getName()
232 			);
233 		} else if (ITestResult.class.isInstance(attr)) {
234 			ITestResult result = (ITestResult)attr;
235 			ITestNGMethod method = result.getMethod();
236 			ITestContext context = result.getTestContext();
237 
238 			return String.format(
239 				"ITestResult >> Suite: [%s]. Test: [%s]. Class.Method: [%s].[%s]",
240 				context.getSuite().getName(), context.getName(),
241 				method.getRealClass().getSimpleName(),
242 				method.getMethodName()
243 			);
244 		}
245 
246 		return attr.toString();
247 	}
