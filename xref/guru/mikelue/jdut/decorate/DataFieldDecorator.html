1   package guru.mikelue.jdut.decorate;

3   import java.util.function.BiPredicate;
4   
5   import guru.mikelue.jdut.datagrain.DataField;
6   import guru.mikelue.jdut.datagrain.DataRow;
7   
8   /**
9    * Likes {@link DataFieldDecorator}, with iteration of data fields.
10   */
11  @FunctionalInterface
12  public interface DataFieldDecorator {
13  	/**
14  	 * Turns this decorator to {@link DataGrainDecorator}.
15  	 *
16  	 * @return new instance of decorator of data grain
17  	 */
18  	default DataGrainDecorator toDataGrainDecorator()
19  	{
20  		return rowBuilder -> rowBuilder.getStreamOfFields()
21  			.forEach(
22  				dataField -> decorate(rowBuilder, dataField)
23  			);
24  	}
25  
26  	/**
27  	 * Builds a decorator which executes decoration if testing of <em>dataFieldPredicate</em> is true.
28  	 *
29  	 * @param dataFieldPredicate The predicate of row builder
30  	 *
31  	 * @return new decorator with checking of <em>dataFieldPredicate</em>
32  	 */
33  	default DataFieldDecorator predicate(BiPredicate<DataRow.Builder, DataField<?>> dataFieldPredicate)
34  	{
35  		return (rowBuilder, dataField) -> {
36  			if (dataFieldPredicate.test(rowBuilder, dataField)) {
37  				decorate(rowBuilder, dataField);
38  			}
39  		};
40  	}
41  
42  	/**
43  	 * Iterates the data field.
44  	 *
45  	 * @param rowBuilder used to modify the data of row
46  	 * @param dataField iterated field currently
47  	 */
48  	public void decorate(DataRow.Builder rowBuilder, DataField<?> dataField);
49  }
