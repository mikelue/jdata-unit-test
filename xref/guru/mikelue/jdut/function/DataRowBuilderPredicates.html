11  /**
12   * Buildings for {@link Predicate} of {@link DataRow.Builder}.<br>
13   *
14   * <p>These labmda is used with {@link DataGrainDecorator#predicate}.</p>
15   */
16  public final class DataRowBuilderPredicates {
17  	private DataRowBuilderPredicates() {}
18  
19  	/**
20  	 * Builds {@link Predicate} of a not-existing column on table.
21  	 *
22  	 * @param tableName The name of table
23  	 * @param columnName the name of column
24  	 *
25  	 * @return The predicate for row builder
26  	 */
27  	static public Predicate<DataRow.Builder> notExistingColumn(String tableName, String columnName)
28  	{
29  		String safeTableName = StringUtils.trimToNull(tableName);
30  		String safeColumnName = StringUtils.trimToNull(columnName);
31  
32  		Validate.notNull(safeTableName, "Need viable name of table");
33  		Validate.notNull(safeColumnName, "Need viable name of column");
34  
35  		return rowBuilder -> safeTableName.equals(rowBuilder.getTableName()) &&
36  			!rowBuilder.getDataField(safeColumnName).isPresent();
37  	}
38  }
