5   import org.apache.commons.lang3.StringUtils;
6   import org.apache.commons.lang3.Validate;

8   import guru.mikelue.jdut.datagrain.DataField;

10  /**
11   * Buildings for {@link Predicate} of {@link DataField}.
12   */
13  public final class DataFieldPredicates {
14  	private DataFieldPredicates() {}
15  
16  	/**
17  	 * Builds {@link Predicate} of null value for a column on table.
18  	 *
19  	 * @param <T> The type of data field
20  	 * @param tableName The name of table
21  	 * @param columnName the name of column
22  	 *
23  	 * @return The predicate of data field
24  	 */
25  	static <T> Predicate<DataField<T>> nullValue(String tableName, String columnName)
26  	{
27  		String safeTableName = StringUtils.trimToNull(tableName);
28  		String safeColumnName = StringUtils.trimToNull(columnName);
29  
30  		Validate.notNull(safeTableName, "Need viable name of table");
31  		Validate.notNull(safeColumnName, "Need viable name of column");
32  
33  		return dataField -> dataField.getData() == null &&
34  			safeTableName.equals(dataField.getTableName()) &&
35  			safeColumnName.equals(dataField.getColumnName());
36  	}
37  
38  	/**
39  	 * The implementation of {@link Predicate Predicate}({@code DataField<?>}).
40  	 *
41  	 * @param dataField The tested data field
42  	 *
43  	 * @return true if the data of the field is not set by supplier
44  	 */
45  	static boolean nonSupplier(DataField<?> dataField)
46  	{
47  		return !dataField.getDataSupplier()
48  			.isPresent();
49  	}
50  }
