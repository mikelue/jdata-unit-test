1   package guru.mikelue.jdut;

3   import java.sql.Connection;
4   import java.util.Optional;
5   
6   /**
7    * Used to be access by thread-local variable of current execution of conductor.
8    */
9   public final class ConductorContext {
10  	private static ThreadLocal<Connection> connOfThreadLocal = new ThreadLocal<>();
11  
12  	private ConductorContext() {}
13  
14  	/**
15  	 * Gets current connection.
16  	 *
17  	 * @return current object of connection, my be empty
18  	 */
19  	public static Optional<Connection> getCurrentConnection()
20  	{
21  		return Optional.ofNullable(connOfThreadLocal.get());
22  	}
23  
24  	/**
25  	 * Sets current connection.
26  	 *
27  	 * @param connection The connection object to be keeped in thread local
28  	 */
29  	static void setCurrentConnection(Connection connection)
30  	{
31  		connOfThreadLocal.set(connection);
32  	}
33  
34  	/**
35  	 * Cleans the current connection.
36  	 */
37  	static void cleanCurrentConnection()
38  	{
39  		connOfThreadLocal.remove();
40  	}
41  }
